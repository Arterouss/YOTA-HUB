2. ACTIVITY DIAGRAM GABUNGAN (SEMINAR & E-LEARNING)
------------------------------------------------------
@startuml
start
:Lihat & Daftar Program (Seminar / E-Learning);

if (Validasi Status/Kuota?) then (Penuh/Tutup)
  :Tampil Error + Rollback;
  stop
endif

:INSERT Data Pendaftaran (seminar_user / course_enrollments);

if (Harga > 0 / Berbayar?) then (Ya)
  :payment_status = pending;
  :Tunggu Verifikasi Pembayaran Admin;
  :payment_status = paid;
else (Tidak/Gratis)
  :payment_status = paid;
endif

if (Jenis Program?) then (Seminar / Webinar)
  :Ikuti Acara Sesuai Jadwal;
else (E-Learning)
  repeat
    :Nonton Video / Baca Modul;
    :Update module_progress = completed;
    :Update progress_percentage;
  repeat while (Progress < 100%?) is (Belum Selesai)
endif

:status = completed;

if (grading_type == 'manual') then (Ya)
  :Wajib Mengumpulkan Tugas / Resume;
  :Admin Menilai & Konfirmasi;
else (auto)
  if (Jenis Program?) then (Seminar)
    :Isi Kuis & Form Evaluasi;
  else (E-Learning)
    :Penilaian Selesai Otomatis;
  endif
endif

:Klaim XP / Poin Reward;
:Generate certificate_code;
:Unduh Sertifikat Tersedia;
stop
@enduml

4. SEQUENCE DIAGRAM GABUNGAN (Pendaftaran & Belajar)
------------------------------------------------------
@startuml
actor Member
participant "ProgramController" as Ctrl
participant "AdminController" as Admin
database DB

Member -> Ctrl : Request Daftar Program (Seminar / E-Learning)
Ctrl -> DB : BEGIN TRANSACTION
Ctrl -> DB : SELECT FOR UPDATE (lockForUpdate)
alt Kuota Habis
  Ctrl --> Member : Error Kuota Penuh + ROLLBACK
else Kuota Tersedia
  Ctrl -> DB : INSERT Data Pendaftaran
  Ctrl -> DB : UPDATE Sisa Kuota
  Ctrl -> DB : COMMIT
  
  alt Jika Program Berbayar (Harga > 0)
    Ctrl -> DB : UPDATE payment_status = 'pending'
    Ctrl --> Member : Success (Menunggu Verifikasi)
    Admin -> DB : Verifikasi Pembayaran (Set 'paid')
    Admin --> Member : Akses Terbuka
  else Jika Program Gratis
    Ctrl -> DB : UPDATE payment_status = 'paid'
    Ctrl --> Member : Success (Akses Langsung Buka)
  end
end

== Proses Belajar & Penilaian ==

alt Jika Program = Seminar
  Member -> Ctrl : Menghadiri Seminar (Absen/Selesai)
  alt grading_type == 'manual'
    Member -> Ctrl : Submit Link Tugas/Resume
    Admin -> DB : Verifikasi Tugas & Set Lulus
  else grading_type == 'auto'
    Member -> Ctrl : Submit Form Kuis & Evaluasi
    Ctrl -> DB : Set Lulus
  end

else Jika Program = E-Learning
  loop Hingga Progress 100%
    Member -> Ctrl : Tonton Video & Selesai Modul
    Ctrl -> DB : UPDATE module_progress & percentage
  end
  
  alt grading_type == 'manual'
    Member -> Ctrl : Submit Tugas Praktik E-Learning
    Admin -> DB : Verifikasi Tugas & Set Lulus
  else grading_type == 'auto'
    Ctrl -> DB : Set Lulus Otomatis
  end
end

== Klaim Poin & Sertifikat ==
Ctrl -> DB : Tambah Experience Points (XP)
Ctrl -> DB : Generate certificate_code
Ctrl --> Member : Unduh Sertifikat Tersedia
@enduml
